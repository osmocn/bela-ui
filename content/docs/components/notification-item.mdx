## Usage

```tsx
import { NotificationItem } from "@/components/bela-ui/notification-item";

export default function Example() {
  return (
    <NotificationItem
      notification={{
        id: "1",
        title: "New comment",
        description: "Sarah commented on your post.",
        createdAt: new Date(),
        read: false,
      }}
    />
  );
}
```

## Props

### NotificationItem

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `notification` | `Notification` | `undefined` | Notification object to render |
| `className` | `string` | `undefined` | Additional CSS classes |

### Notification

| Property | Type | Default | Description |
| --- | --- | --- | --- |
| `id` | `string` | `undefined` | Unique notification identifier |
| `title` | `string` | `undefined` | Notification title |
| `description` | `string` | `undefined` | Additional description text |
| `createdAt` | `Date` | `undefined` | Notification timestamp |
| `read` | `boolean` | `false` | Whether the notification has been read |
